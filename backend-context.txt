import {NextResponse} from 'next/server'
import {randomUUID} from 'node:crypto'

import {writeClient} from '@/sanity/lib/writeClient'

export const runtime = 'nodejs'

const MAX_PHOTOS = 5
const MAX_FILE_SIZE = 10 * 1024 * 1024

const ALLOWED_IMAGE_TYPES = new Set([
  'image/jpeg',
  'image/png',
  'image/webp',
  'image/heic',
  'image/heif',
])

type UploadedPhoto = {
  _key: string
  _type: 'image'
  asset: {
    _type: 'reference'
    _ref: string
  }
}

function getRequiredText(
  formData: FormData,
  fieldName: string,
  maxLength: number,
): string {
  const value = formData.get(fieldName)

  if (typeof value !== 'string') {
    throw new Error(`Поле «${fieldName}» не заполнено`)
  }

  const normalizedValue = value.trim()

  if (!normalizedValue) {
    throw new Error(`Поле «${fieldName}» не заполнено`)
  }

  if (normalizedValue.length > maxLength) {
    throw new Error(
      `Поле «${fieldName}» не должно превышать ${maxLength} символов`,
    )
  }

  return normalizedValue
}

function getOptionalText(
  formData: FormData,
  fieldName: string,
  maxLength: number,
): string | undefined {
  const value = formData.get(fieldName)

  if (typeof value !== 'string') {
    return undefined
  }

  const normalizedValue = value.trim()

  if (!normalizedValue) {
    return undefined
  }

  if (normalizedValue.length > maxLength) {
    throw new Error(
      `Поле «${fieldName}» не должно превышать ${maxLength} символов`,
    )
  }

  return normalizedValue
}

function getCoordinate(
  formData: FormData,
  fieldName: 'latitude' | 'longitude',
): number {
  const rawValue = formData.get(fieldName)

  if (typeof rawValue !== 'string' || !rawValue.trim()) {
    throw new Error('Необходимо указать точку нарушения на карте')
  }

  const coordinate = Number(rawValue)

  if (!Number.isFinite(coordinate)) {
    throw new Error('Переданы некорректные координаты')
  }

  if (fieldName === 'latitude' && (coordinate < -90 || coordinate > 90)) {
    throw new Error('Передана некорректная широта')
  }

  if (fieldName === 'longitude' && (coordinate < -180 || coordinate > 180)) {
    throw new Error('Передана некорректная долгота')
  }

  return coordinate
}

function validateEmail(email: string | undefined): void {
  if (!email) {
    return
  }

  const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/

  if (!emailPattern.test(email)) {
    throw new Error('Укажите корректный адрес электронной почты')
  }
}

function validateFiles(files: File[]): void {
  if (files.length > MAX_PHOTOS) {
    throw new Error(`Можно загрузить не более ${MAX_PHOTOS} фотографий`)
  }

  for (const file of files) {
    if (!ALLOWED_IMAGE_TYPES.has(file.type)) {
      throw new Error(
        `Файл «${file.name}» имеет неподдерживаемый формат изображения`,
      )
    }

    if (file.size === 0) {
      throw new Error(`Файл «${file.name}» пустой`)
    }

    if (file.size > MAX_FILE_SIZE) {
      throw new Error(
        `Размер файла «${file.name}» превышает 10 мегабайт`,
      )
    }
  }
}

async function uploadPhotos(files: File[]): Promise<UploadedPhoto[]> {
  const uploadedPhotos: UploadedPhoto[] = []

  for (const file of files) {
    const arrayBuffer = await file.arrayBuffer()
    const buffer = Buffer.from(arrayBuffer)

    const asset = await writeClient.assets.upload('image', buffer, {
      filename: file.name,
      contentType: file.type,
    })

    uploadedPhotos.push({
      _key: randomUUID(),
      _type: 'image',
      asset: {
        _type: 'reference',
        _ref: asset._id,
      },
    })
  }

  return uploadedPhotos
}

async function createComplaintId(): Promise<string> {
  const latestComplaintId = await writeClient.fetch<string | null>(
    `
      *[
        _type == "complaint" &&
        defined(complaintId)
      ]
      | order(createdAt desc)[0].complaintId
    `,
  )

  const latestNumberMatch = latestComplaintId?.match(/^GF-(\d+)$/)
  const latestNumber = latestNumberMatch
    ? Number(latestNumberMatch[1])
    : 0

  const nextNumber = latestNumber + 1

  return `GF-${String(nextNumber).padStart(6, '0')}`
}

export async function POST(request: Request) {
  try {
    if (!process.env.SANITY_API_WRITE_TOKEN) {
      console.error('SANITY_API_WRITE_TOKEN is not configured')

      return NextResponse.json(
        {
          success: false,
          error: 'Сервер не настроен для приёма жалоб',
        },
        {
          status: 500,
        },
      )
    }

    const formData = await request.formData()

    const title = getRequiredText(formData, 'title', 150)
    const description = getRequiredText(formData, 'description', 5000)
    const category = getRequiredText(formData, 'category', 100)
    const region = getRequiredText(formData, 'region', 150)
    const address = getRequiredText(formData, 'address', 500)

    const phone = getOptionalText(formData, 'phone', 50)
    const email = getOptionalText(formData, 'email', 254)

    validateEmail(email)

    if (!phone && !email) {
      throw new Error(
        'Укажите хотя бы один способ связи: телефон или электронную почту',
      )
    }

    const latitude = getCoordinate(formData, 'latitude')
    const longitude = getCoordinate(formData, 'longitude')

    const files = formData
      .getAll('photos')
      .filter((value): value is File => value instanceof File)

    validateFiles(files)

    const photos = await uploadPhotos(files)
    const complaintId = await createComplaintId()
    const createdAt = new Date().toISOString()

    const complaint = await writeClient.create({
      _type: 'complaint',
      complaintId,
      title,
      description,
      category,
      region,
      address,
      latitude,
      longitude,
      phone,
      email,
      photos,
      status: 'moderation',
      createdAt,
    })

    return NextResponse.json(
      {
        success: true,
        complaintId,
        documentId: complaint._id,
        message: 'Жалоба отправлена на модерацию',
      },
      {
        status: 201,
      },
    )
  } catch (error) {
    console.error('Complaint submission error:', error)

    const message =
      error instanceof Error
        ? error.message
        : 'Не удалось отправить жалобу'

    return NextResponse.json(
      {
        success: false,
        error: message,
      },
      {
        status: 400,
      },
    )
  }
}

export const videoType = defineType({

  name: 'video',

  title: 'Видео',

  type: 'document',

  fields: [

    defineField({
      name: 'slug',
      title: 'URL адрес',
      type: 'slug',
      options: {
        source: 'title',
        maxLength: 96,
      },
    }),


    defineField({
      name: 'title',
      title: 'Название видео',
      type: 'string',
    }),


    defineField({
      name: 'description',
      title: 'Описание',
      type: 'text',
    }),


    defineField({
      name: 'youtubeUrl',
      title: 'Ссылка YouTube',
      type: 'url',
    }),


    defineField({
      name: 'date',
      title: 'Дата публикации',
      type: 'date',
    }),


    defineField({
      name: 'cover',
      title: 'Обложка',
      type: 'image',
      options: {
        hotspot: true,
      },
    }),

  ],

})

import {StructureResolver} from 'sanity/structure'

export const structure: StructureResolver = (S) =>
  S.list()
    .title('Контент')
    .items([
      S.documentTypeListItem('video').title('Видео'),

      S.documentTypeListItem('article').title('Статьи'),

      S.documentTypeListItem('post').title('Публикации'),

      S.documentTypeListItem('contact').title('Контакты'),

      S.divider(),

      S.documentTypeListItem('complaint').title('Жалобы граждан'),
    ])
